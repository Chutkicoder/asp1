using System; class ArmstrongNumber { static void Main() { Console.Write("Enter a number: "); int num = int.Parse(Console.ReadLine()); int temp = num, sum = 0, digit; while (num > 0) { digit = num % 10; sum += digit * digit * digit; num = num / 10; } if (temp == sum) Console.WriteLine("The number is an Armstrong number."); else Console.WriteLine("The number is NOT an Armstrong number."); Console.ReadKey(); } }
